class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        map<char,int> m;
        int start=1;
        int res=0;
        for(int i=1;i<=s.size();i++){
            char c = s[i-1];
            if(m[c]>=start){
                start = m[c] + 1;
                m[c] = i;
            }
            else{
                m[c] = i;
                res = max(res,i-start+1);
            }
        }
        return res;
    }
};
